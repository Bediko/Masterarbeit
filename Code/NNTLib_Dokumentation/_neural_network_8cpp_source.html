125 void NeuralNetwork::CalculateMSE(const DataContainer* data)

173 double NeuralNetwork::GenerateRandomWeight(int inputVectorCountWithBias)
174 {
175  switch (WeightInitType)
176  {
177  case WeightInitEnum::UNIFORM:
178  {
179  std::uniform_real_distribution<double> dist(-1.0, 1.0);
180  return dist(generator);
181  }
182  case WeightInitEnum::LECUN:
183  {
184  double sigma = pow(inputVectorCountWithBias,-0.5);
185  std::normal_distribution<double> dist(0.0,sigma);
186  return dist(generator);
187  }
188  case WeightInitEnum::NONE:
189  // default:
190  return 0;
191  }
192 
193  return 0;
194 }

334 void NeuralNetwork::Propagate(const double *input)
335 {
336  for(int i=0;i<Layers[0].InputVectorCount;++i)
337  {
338  //inputlayer hat keine neuronen daher daten in input vektor des ersten layers mit Neuronen übernehmen
339  Layers[0].InputVector[i] = input[i];
340  }
341 
342  int k,j;
343  double net;
344  Neuron *neuron;
345 
346  for(int i=0;i<LayersCount;++i)
347  {
348  for(k=0;k<Layers[i].NeuronCount;++k)
349  {
350  neuron=&Layers[i].Neurons[k];
351  net=0;
352 
353  //summiere gewichte * output des vorherigen layer auf => net
354 
355  //unrolling des loops beschleunigte das propagieren um Faktor ~2 zumindestens beim windows compiler
356 
357  /*for(j=0;j<Layers[i].InputVectorCount;++j)
358  {
359  net+=neuron->Weights[j] * Layers[i].InputVector[j];
360  }*/
361 
362  //unrolling
363  j = Layers[i].InputVectorCount & 3; //das gleiche wie module 4
364 
365  //behandle Neuronen die nicht in der nachfolgenden 4er Schritt loop behandelt werden können
366  switch (j)
367  {
368  case 3:
369  net+=neuron->Weights[2] * Layers[i].InputVector[2];
370 
371  case 2:
372  net+=neuron->Weights[1] * Layers[i].InputVector[1];
373 
374  case 1:
375  net+=neuron->Weights[0] * Layers[i].InputVector[0];
376  case 0:
377  break;
378  }
379 
380  for(; j != Layers[i].InputVectorCount; j += 4)
381  {
382  net +=
383  neuron->Weights[j] * Layers[i].InputVector[j]+
384  neuron->Weights[j+1] * Layers[i].InputVector[j+1]+
385  neuron->Weights[j+2] * Layers[i].InputVector[j+2]+
386  neuron->Weights[j+3] * Layers[i].InputVector[j+3];
387  }
388 
389  net+= neuron->Weights[Layers[i].InputVectorCount] ;
390  //unrolling ende
391 
392  //aktivierungsfunktion f_net(net) auf net anwenden und in ausgabe des neurons speichern
393  neuron->Output = ActivationFunction(FunctionType,net);
394 
395  //ausgaben des layers als input des nächsten layers verwenden
396  if(i < LayersCount-1)//letzter layer nicht weiter nach vorne kopieren
397  Layers[i + 1].InputVector[k]=neuron->Output;
398  }
399  }
400 }
